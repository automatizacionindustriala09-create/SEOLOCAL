function dashboardHashPassword(password = '') {
  return `sha256:${crypto.createHash('sha256').update(String(password)).digest('hex')}`;
}

function dashboardHashToken(token = '') {
  return crypto.createHash('sha256').update(String(token)).digest('hex');
}

function normalizeDashboardBoolean(value) {
  return value === true || value === 'true' || value === 1 || value === '1';
}

function normalizeDashboardNumber(value, fallback = null) {
  if (value === '' || value === null || value === undefined) return fallback;
  const parsed = Number(value);
  return Number.isFinite(parsed) ? parsed : fallback;
}

function normalizeDashboardSlug(value = '') {
  return String(value || '')
    .normalize('NFD')
    .replace(/[\u0300-\u036f]/g, '')
    .toLowerCase()
    .replace(/[^a-z0-9]+/g, '-')
    .replace(/^-+|-+$/g, '') || `item-${Date.now()}`;
}

async function getDashboardPermissions(roleCode) {
  const result = await pool.query(`
    SELECT permission_code
    FROM seo_local_dashboard_role_permission
    WHERE role_code = $1
    ORDER BY permission_code
  `, [roleCode]);
  return result.rows.map((row) => row.permission_code);
}

async function logDashboardAction(client, userId, action, model, resId, payload = {}) {
  await client.query(`
    INSERT INTO seo_local_dashboard_audit_log(user_id, action, model, res_id, payload)
    VALUES ($1, $2, $3, $4, $5::jsonb)
  `, [userId || null, action, model, resId || null, JSON.stringify(payload || {})]);
}

async function requireDashboardUser(request, response, next) {
  try {
    const header = request.headers.authorization || '';
    const token = header.startsWith('Bearer ') ? header.slice(7).trim() : '';
    if (!token) return response.status(401).json({ error: 'Sesión requerida.' });

    const tokenHash = dashboardHashToken(token);
    const result = await pool.query(`
      SELECT u.id, u.login, u.role_code AS base_role_code,
             COALESCE(u.dashboard_role_code, u.role_code) AS dashboard_role_code,
             u.active, u.partner_id, u.agency_partner_id, p.name, r.name AS role_name,
             s.id AS session_id
      FROM seo_local_dashboard_session s
      JOIN res_users u ON u.id = s.user_id
      LEFT JOIN res_partner p ON p.id = u.partner_id
      LEFT JOIN seo_local_dashboard_role r ON r.role_code = COALESCE(u.dashboard_role_code, u.role_code)
      WHERE s.token_hash = $1 AND s.active = TRUE AND s.expires_at > NOW() AND u.active = TRUE
      LIMIT 1
    `, [tokenHash]);

    if (!result.rowCount) return response.status(401).json({ error: 'Sesión inválida o vencida.' });

    const user = result.rows[0];
    const permissions = await getDashboardPermissions(user.dashboard_role_code);
    await pool.query('UPDATE seo_local_dashboard_session SET last_seen_at = NOW() WHERE id = $1', [user.session_id]);

    request.dashboardUser = {
      id: user.id,
      login: user.login,
      name: user.name || user.login,
      baseRole: user.base_role_code,
      roleCode: user.dashboard_role_code,
      roleName: user.role_name || user.dashboard_role_code,
      partnerId: user.partner_id,
      agencyPartnerId: user.agency_partner_id,
      permissions,
    };
    return next();
  } catch (error) {
    return next(error);
  }
}

function can(request, permission) {
  return request.dashboardUser?.permissions?.includes(permission);
}

function requirePermission(...permissions) {
  return (request, response, next) => {
    if (permissions.some((permission) => can(request, permission))) return next();
    return response.status(403).json({ error: `Permisos insuficientes. Requiere: ${permissions.join(' o ')}` });
  };
}

function isAgencyScoped(request) {
  return request.dashboardUser?.roleCode === 'agency_manager';
}

function scopedAgencyId(request) {
  return request.dashboardUser?.agencyPartnerId || null;
}

function assertAgencyScope(request, agencyId) {
  if (!isAgencyScoped(request)) return;
  if (!scopedAgencyId(request) || Number(scopedAgencyId(request)) !== Number(agencyId)) {
    const error = new Error('Este rol solo puede operar sobre su propia agencia.');
    error.status = 403;
    throw error;
  }
}

function serializeUser(user) {
  return {
    id: user.id,
    login: user.login,
    name: user.name || user.login,
    baseRole: user.baseRole || user.base_role_code,
    roleCode: user.roleCode || user.dashboard_role_code,
    roleName: user.roleName || user.role_name || user.dashboard_role_code,
    agencyPartnerId: user.agencyPartnerId || user.agency_partner_id || null,
    permissions: user.permissions || [],
  };
}

app.post('/api/v1/admin/auth/login', asyncRoute(async (request, response) => {
  const login = String(request.body?.login || '').trim().toLowerCase();
  const password = String(request.body?.password || '');
  if (!login || !password) return response.status(422).json({ error: 'Usuario y contraseña son obligatorios.' });

  const result = await pool.query(`
    SELECT u.id, u.login, u.password_hash, u.role_code AS base_role_code,
           COALESCE(u.dashboard_role_code, u.role_code) AS dashboard_role_code,
           u.active, u.partner_id, u.agency_partner_id, p.name, r.name AS role_name
    FROM res_users u
    LEFT JOIN res_partner p ON p.id = u.partner_id
    LEFT JOIN seo_local_dashboard_role r ON r.role_code = COALESCE(u.dashboard_role_code, u.role_code)
    WHERE LOWER(u.login) = LOWER($1)
    LIMIT 1
  `, [login]);

  if (!result.rowCount || !result.rows[0].active) return response.status(401).json({ error: 'Credenciales inválidas.' });

  const user = result.rows[0];
  const expected = user.password_hash || '';
  if (expected && expected !== dashboardHashPassword(password)) return response.status(401).json({ error: 'Credenciales inválidas.' });

  const permissions = await getDashboardPermissions(user.dashboard_role_code);
  if (!permissions.length) return response.status(403).json({ error: 'El usuario no tiene permisos de dashboard.' });

  const rawToken = crypto.randomBytes(32).toString('hex');
  const tokenHash = dashboardHashToken(rawToken);
  await pool.query(`
    INSERT INTO seo_local_dashboard_session(user_id, token_hash, expires_at, user_agent, ip_address)
    VALUES ($1, $2, NOW() + INTERVAL '12 hours', $3, $4)
  `, [user.id, tokenHash, String(request.headers['user-agent'] || ''), request.ip || '']);

  await pool.query('UPDATE res_users SET last_login = NOW(), last_dashboard_login = NOW(), write_date = NOW() WHERE id = $1', [user.id]);

  response.json({
    token: rawToken,
    user: serializeUser({ ...user, permissions }),
  });
}));

  const result = await pool.query(`
    SELECT p.permission_code, p.module_code, p.name, p.description,

  const agencyParams = isAgencyScoped(request) ? [scopedAgencyId(request)] : [];
  const result = await pool.query(`
    SELECT
      (SELECT COUNT(*)::int FROM seo_local_agency_profile ${agencyFilter}) AS total_agencies,
      (SELECT COUNT(*)::int FROM seo_local_agency_profile ${agencyFilter ? agencyFilter + " AND status = 'published'" : "WHERE status = 'published'"}) AS published_agencies,

      (SELECT COUNT(*)::int FROM seo_local_review) AS total_reviews,
      (SELECT COUNT(*)::int FROM seo_local_subscription_plan WHERE active = TRUE) AS total_plans,
      COALESCE((SELECT ROUND(AVG(rating)::numeric, 2) FROM seo_local_agency_profile ${agencyFilter}), 0) AS average_rating,
      COALESCE((SELECT SUM(expected_revenue) FROM crm_lead WHERE active = TRUE), 0) AS pipeline_value
  `, agencyParams);
  const row = result.rows[0];
  response.json({
    totalAgencies: row.total_agencies,
    publishedAgencies: row.published_agencies,
    totalServices: row.total_services,
    totalCategories: row.total_categories,
    totalLeads: row.total_leads,
    totalReviews: row.total_reviews,
    totalPlans: row.total_plans,
    averageRating: Number(row.average_rating),
    pipelineValue: Number(row.pipeline_value),
  });
}));

app.get('/api/v1/admin/agencies', requireDashboardUser, requirePermission('agencies.read'), asyncRoute(async (request, response) => {
  const params = [];
  let where = '';
  if (isAgencyScoped(request)) {
    where = 'WHERE ap.partner_id = $1';
    params.push(scopedAgencyId(request));
  }
  const result = await pool.query(`
    SELECT ap.partner_id AS id, p.name, p.email, p.phone, p.website, p.city, p.country_code,
           ap.summary, ap.highlight_review, ap.rating, ap.reviews_count, ap.starting_price,
           ap.is_verified, ap.is_top_rated, ap.status, ap.logo_letter, ap.image_url,
           d.tagline, d.focus, d.methodology, d.client_profile, d.promise_headline,
           d.industries, d.identity_tags, s.status AS subscription_status, sp.name AS plan_name, ap.write_date
    FROM seo_local_agency_profile ap
    JOIN res_partner p ON p.id = ap.partner_id
    LEFT JOIN seo_local_agency_profile_detail d ON d.agency_partner_id = ap.partner_id
    LEFT JOIN LATERAL (
      SELECT * FROM seo_local_agency_subscription ss WHERE ss.agency_partner_id = ap.partner_id ORDER BY ss.starts_at DESC, ss.id DESC LIMIT 1
    ) s ON TRUE
    LEFT JOIN seo_local_subscription_plan sp ON sp.id = s.plan_id
    ${where}
    ORDER BY ap.is_top_rated DESC, ap.rating DESC, p.name
  `, params);
  response.json({ items: result.rows.map((row) => ({ ...row, rating: Number(row.rating), starting_price: Number(row.starting_price) })) });
}));

  const id = Number(request.params.id);
  const module = String(request.params.module || '');
  assertAgencyScope(request, id);
  const b = request.body || {};
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    if (module === 'detail') {
      await client.query(`
        INSERT INTO seo_local_agency_profile_detail(agency_partner_id, tagline, focus, methodology, industries, client_profile, identity_tags, promise_headline, active)
        VALUES($1,COALESCE($2,''),COALESCE($3,''),COALESCE($4,''),COALESCE($5::jsonb,'[]'::jsonb),COALESCE($6,''),COALESCE($7::jsonb,'[]'::jsonb),COALESCE($8,''),COALESCE($9,TRUE))
        ON CONFLICT(agency_partner_id) DO UPDATE SET tagline=EXCLUDED.tagline, focus=EXCLUDED.focus, methodology=EXCLUDED.methodology,
          industries=EXCLUDED.industries, client_profile=EXCLUDED.client_profile, identity_tags=EXCLUDED.identity_tags,

      `, [id, b.tagline, b.focus, b.methodology, JSON.stringify(b.industries || []), b.client_profile, JSON.stringify(b.identity_tags || []), b.promise_headline, b.active]);
    } else {
      const tableMap = {
        channels: 'seo_local_agency_channel',
        hours: 'seo_local_agency_business_hour',
        team: 'seo_local_agency_team_member',
        certifications: 'seo_local_agency_certification',
      };
      const table = tableMap[module];
      if (!table) return response.status(422).json({ error: 'Módulo inválido.' });
      const rows = Array.isArray(b) ? b : [];
      await client.query(`DELETE FROM ${table} WHERE agency_partner_id=$1`, [id]);
      for (const [idx, row] of rows.entries()) {
        if (module === 'channels') await client.query(`INSERT INTO seo_local_agency_channel(agency_partner_id, channel_type, label, value, url, is_verified, sequence, active) VALUES($1,$2,$3,$4,$5,COALESCE($6,TRUE),$7,COALESCE($8,TRUE))`, [id, row.channel_type || row.type || 'website', row.label || 'Canal', row.value || '', row.url || '', row.is_verified, row.sequence || idx + 1, row.active]);
        if (module === 'hours') await client.query(`INSERT INTO seo_local_agency_business_hour(agency_partner_id, day_label, opens_at, closes_at, is_closed, sequence) VALUES($1,$2,$3,$4,COALESCE($5,FALSE),$6)`, [id, row.day_label || 'Día', row.opens_at || '', row.closes_at || '', row.is_closed, row.sequence || idx + 1]);
        if (module === 'team') await client.query(`INSERT INTO seo_local_agency_team_member(agency_partner_id, name, role_title, bio, avatar_url, specialty, sequence, active) VALUES($1,$2,$3,$4,$5,$6,$7,COALESCE($8,TRUE))`, [id, row.name || 'Miembro', row.role_title || row.role || 'Especialista', row.bio || '', row.avatar_url || '', row.specialty || '', row.sequence || idx + 1, row.active]);
        if (module === 'certifications') await client.query(`INSERT INTO seo_local_agency_certification(agency_partner_id, issuer, title, credential_url, valid_until, sequence, active) VALUES($1,$2,$3,$4,NULLIF($5,'')::date,$6,COALESCE($7,TRUE))`, [id, row.issuer || 'Certificador', row.title || 'Certificación', row.credential_url || '', row.valid_until || null, row.sequence || idx + 1, row.active]);
      }
    }
    await logDashboardAction(client, request.dashboardUser.id, 'update_module', `seo_local_agency_${module}`, id, b);
    await client.query('COMMIT');
    response.json({ ok: true, id, module });
  } catch (error) { await client.query('ROLLBACK'); throw error; } finally { client.release(); }
}));
