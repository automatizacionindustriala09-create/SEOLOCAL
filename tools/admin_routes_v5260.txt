function dashboardHashPassword(password = '') {
  return `sha256:${crypto.createHash('sha256').update(String(password)).digest('hex')}`;
}

function dashboardHashToken(token = '') {
  return crypto.createHash('sha256').update(String(token)).digest('hex');
}

function normalizeDashboardBoolean(value) {
  return value === true || value === 'true' || value === 1 || value === '1';
}

async function logDashboardAction(client, userId, action, model, resId, payload = {}) {
  await client.query(`
    INSERT INTO seo_local_dashboard_audit_log(user_id, action, model, res_id, payload)
    VALUES ($1, $2, $3, $4, $5::jsonb)
  `, [userId || null, action, model, resId || null, JSON.stringify(payload || {})]);
}

async function requireDashboardUser(request, response, next) {
  try {
    const header = request.headers.authorization || '';
    const token = header.startsWith('Bearer ') ? header.slice(7).trim() : '';
    if (!token) return response.status(401).json({ error: 'Sesión requerida.' });
    const tokenHash = dashboardHashToken(token);
    const result = await pool.query(`
      SELECT u.id, u.login, u.role_code, u.active, p.name, s.id AS session_id
      FROM seo_local_dashboard_session s
      JOIN res_users u ON u.id = s.user_id
      LEFT JOIN res_partner p ON p.id = u.partner_id
      WHERE s.token_hash = $1 AND s.active = TRUE AND s.expires_at > NOW() AND u.active = TRUE
      LIMIT 1
    `, [tokenHash]);
    if (!result.rowCount) return response.status(401).json({ error: 'Sesión inválida o vencida.' });
    await pool.query('UPDATE seo_local_dashboard_session SET last_seen_at = NOW() WHERE id = $1', [result.rows[0].session_id]);
    request.dashboardUser = result.rows[0];
    return next();
  } catch (error) {
    return next(error);
  }
}

function requireDashboardRole(...roles) {
  return (request, response, next) => {
    const role = request.dashboardUser?.role_code;
    if (!role || (roles.length && !roles.includes(role))) return response.status(403).json({ error: 'Permisos insuficientes.' });
    next();
  };
}

app.post('/api/v1/admin/auth/login', asyncRoute(async (request, response) => {
  const login = String(request.body?.login || '').trim().toLowerCase();
  const password = String(request.body?.password || '');
  if (!login || !password) return response.status(422).json({ error: 'Usuario y contraseña son obligatorios.' });

  const result = await pool.query(`
    SELECT u.id, u.login, u.password_hash, u.role_code, u.active, p.name
    FROM res_users u
    LEFT JOIN res_partner p ON p.id = u.partner_id
    WHERE LOWER(u.login) = LOWER($1)
    LIMIT 1
  `, [login]);

  if (!result.rowCount || !result.rows[0].active) return response.status(401).json({ error: 'Credenciales inválidas.' });
  const user = result.rows[0];
  const expected = user.password_hash || '';
  if (expected && expected !== dashboardHashPassword(password)) return response.status(401).json({ error: 'Credenciales inválidas.' });
  if (!['superadmin', 'admin', 'agency', 'support'].includes(user.role_code)) return response.status(403).json({ error: 'El usuario no tiene rol de dashboard.' });

  const rawToken = crypto.randomBytes(32).toString('hex');
  const tokenHash = dashboardHashToken(rawToken);
  await pool.query(`
    INSERT INTO seo_local_dashboard_session(user_id, token_hash, expires_at, user_agent, ip_address)
    VALUES ($1, $2, NOW() + INTERVAL '12 hours', $3, $4)
  `, [user.id, tokenHash, String(request.headers['user-agent'] || ''), request.ip || '']);
  await pool.query('UPDATE res_users SET last_login = NOW(), write_date = NOW() WHERE id = $1', [user.id]);

  response.json({
    token: rawToken,
    user: { id: user.id, login: user.login, name: user.name || user.login, roleCode: user.role_code },
  });
}));

app.get('/api/v1/admin/dashboard/summary', requireDashboardUser, asyncRoute(async (_request, response) => {
  const result = await pool.query(`
    SELECT
      (SELECT COUNT(*)::int FROM seo_local_agency_profile) AS total_agencies,
      (SELECT COUNT(*)::int FROM seo_local_agency_profile WHERE status = 'published') AS published_agencies,
      (SELECT COUNT(*)::int FROM product_template WHERE detailed_type = 'service' AND active = TRUE) AS total_services,
      (SELECT COUNT(*)::int FROM seo_local_category WHERE active = TRUE) AS total_categories,
      (SELECT COUNT(*)::int FROM crm_lead WHERE active = TRUE) AS total_leads,
      (SELECT COUNT(*)::int FROM seo_local_review) AS total_reviews,
      (SELECT COUNT(*)::int FROM seo_local_subscription_plan WHERE active = TRUE) AS total_plans,
      COALESCE((SELECT ROUND(AVG(rating)::numeric, 2) FROM seo_local_agency_profile), 0) AS average_rating,
      COALESCE((SELECT SUM(expected_revenue) FROM crm_lead WHERE active = TRUE), 0) AS pipeline_value
  `);
  const row = result.rows[0];
  response.json({
    totalAgencies: row.total_agencies,
    publishedAgencies: row.published_agencies,
    totalServices: row.total_services,
    totalCategories: row.total_categories,
    totalLeads: row.total_leads,
    totalReviews: row.total_reviews,
    totalPlans: row.total_plans,
    averageRating: Number(row.average_rating),
    pipelineValue: Number(row.pipeline_value),
  });
}));

app.get('/api/v1/admin/agencies', requireDashboardUser, asyncRoute(async (_request, response) => {
  const result = await pool.query(`
    SELECT ap.partner_id AS id, p.name, p.email, p.phone, p.website, p.city, p.country_code,
           ap.summary, ap.highlight_review, ap.rating, ap.reviews_count, ap.starting_price,
           ap.is_verified, ap.is_top_rated, ap.status, ap.logo_letter, ap.image_url,
           d.tagline, d.focus, d.methodology, d.client_profile, d.promise_headline,

  `);
  response.json({ items: result.rows.map((row) => ({ ...row, rating: Number(row.rating), starting_price: Number(row.starting_price) })) });
}));

app.put('/api/v1/admin/agencies/:id', requireDashboardUser, requireDashboardRole('superadmin', 'admin', 'agency', 'support'), asyncRoute(async (request, response) => {
  const id = Number(request.params.id);
  if (!Number.isInteger(id) || id <= 0) return response.status(422).json({ error: 'Agencia inválida.' });
  const body = request.body || {};
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    await client.query(`
      UPDATE res_partner SET name = COALESCE(NULLIF($2, ''), name), display_name = COALESCE(NULLIF($2, ''), display_name),
        email = NULLIF($3, ''), phone = NULLIF($4, ''), website = NULLIF($5, ''), city = NULLIF($6, ''), write_date = NOW()
      WHERE id = $1
    `, [id, body.name, body.email, body.phone, body.website, body.city]);
    await client.query(`
      UPDATE seo_local_agency_profile SET summary = COALESCE($2, summary), highlight_review = COALESCE($3, highlight_review),
        rating = COALESCE($4, rating), reviews_count = COALESCE($5, reviews_count), starting_price = COALESCE($6, starting_price),
        is_verified = COALESCE($7, is_verified), is_top_rated = COALESCE($8, is_top_rated), status = COALESCE($9, status), write_date = NOW()
      WHERE partner_id = $1
    `, [id, body.summary, body.highlight_review, body.rating, body.reviews_count, body.starting_price, body.is_verified, body.is_top_rated, body.status]);
    await client.query(`
      INSERT INTO seo_local_agency_profile_detail(agency_partner_id, tagline, focus, methodology, client_profile, promise_headline, industries, identity_tags)
      VALUES ($1, COALESCE($2,''), COALESCE($3,''), COALESCE($4,''), COALESCE($5,''), COALESCE($6,''), COALESCE($7::jsonb, '[]'::jsonb), COALESCE($8::jsonb, '[]'::jsonb))
      ON CONFLICT (agency_partner_id) DO UPDATE SET
        tagline = EXCLUDED.tagline, focus = EXCLUDED.focus, methodology = EXCLUDED.methodology,
        client_profile = EXCLUDED.client_profile, promise_headline = EXCLUDED.promise_headline,
        industries = EXCLUDED.industries, identity_tags = EXCLUDED.identity_tags, write_date = NOW()
    `, [id, body.tagline, body.focus, body.methodology, body.client_profile, body.promise_headline, JSON.stringify(body.industries || []), JSON.stringify(body.identity_tags || [])]);
    await logDashboardAction(client, request.dashboardUser.id, 'update', 'seo_local_agency_profile', id, body);
    await client.query('COMMIT');
    response.json({ ok: true, id });
  } catch (error) { await client.query('ROLLBACK'); throw error; } finally { client.release(); }
}));
