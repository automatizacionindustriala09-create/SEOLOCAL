ALTER TABLE seo_local_agency_profile
  ADD COLUMN IF NOT EXISTS slug VARCHAR(140),
  ADD COLUMN IF NOT EXISTS employees_range VARCHAR(80) NOT NULL DEFAULT 'Equipo certificado',
  ADD COLUMN IF NOT EXISTS experience_years INTEGER NOT NULL DEFAULT 3 CHECK (experience_years >= 0),
  ADD COLUMN IF NOT EXISTS languages JSONB NOT NULL DEFAULT '["Español"]'::jsonb,
  ADD COLUMN IF NOT EXISTS work_modes JSONB NOT NULL DEFAULT '["Remota"]'::jsonb,
  ADD COLUMN IF NOT EXISTS certification_level VARCHAR(40) NOT NULL DEFAULT 'Estándar',
  ADD COLUMN IF NOT EXISTS badge_label VARCHAR(80) NOT NULL DEFAULT 'Estándar',
  ADD COLUMN IF NOT EXISTS recommended BOOLEAN NOT NULL DEFAULT FALSE,
  ADD COLUMN IF NOT EXISTS commercial_summary TEXT NOT NULL DEFAULT '',
  ADD COLUMN IF NOT EXISTS case_study TEXT NOT NULL DEFAULT '',
  ADD COLUMN IF NOT EXISTS response_time_hours INTEGER NOT NULL DEFAULT 24 CHECK (response_time_hours >= 0),
  ADD COLUMN IF NOT EXISTS qualified_projects INTEGER NOT NULL DEFAULT 0 CHECK (qualified_projects >= 0),
  ADD COLUMN IF NOT EXISTS trust_score INTEGER NOT NULL DEFAULT 85 CHECK (trust_score >= 0 AND trust_score <= 100),
  ADD COLUMN IF NOT EXISTS success_rate NUMERIC(5,2) NOT NULL DEFAULT 80 CHECK (success_rate >= 0 AND success_rate <= 100),
  ADD COLUMN IF NOT EXISTS speciality VARCHAR(160) NOT NULL DEFAULT '',
  ADD COLUMN IF NOT EXISTS budget_min NUMERIC(12,2) NOT NULL DEFAULT 0 CHECK (budget_min >= 0),
  ADD COLUMN IF NOT EXISTS budget_max NUMERIC(12,2) NOT NULL DEFAULT 0 CHECK (budget_max >= 0),
  ADD COLUMN IF NOT EXISTS audited BOOLEAN NOT NULL DEFAULT TRUE,
  ADD COLUMN IF NOT EXISTS profile_completeness INTEGER NOT NULL DEFAULT 90 CHECK (profile_completeness >= 0 AND profile_completeness <= 100);

CREATE UNIQUE INDEX IF NOT EXISTS ux_seo_local_agency_profile_slug ON seo_local_agency_profile(slug) WHERE slug IS NOT NULL;

INSERT INTO seo_local_agency_category_rel(agency_partner_id, category_id)
SELECT p.id, c.id
FROM tmp_agency_directory_seed t
JOIN res_partner p ON LOWER(p.email) = LOWER(t.email)
CROSS JOIN LATERAL UNNEST(t.category_external_ids) AS cat(category_external_id)
JOIN seo_local_category c ON c.external_id = cat.category_external_id
ON CONFLICT DO NOTHING;

INSERT INTO seo_local_agency_service_rel(agency_partner_id, product_tmpl_id)
SELECT p.id, pt.id
FROM tmp_agency_directory_seed t
JOIN res_partner p ON LOWER(p.email) = LOWER(t.email)
CROSS JOIN LATERAL UNNEST(t.product_external_ids) AS prod(product_external_id)
JOIN product_template pt ON pt.external_id = prod.product_external_id
ON CONFLICT DO NOTHING;

CREATE OR REPLACE VIEW vw_seo_local_agencies AS
SELECT
  p.id,
  p.name,
  p.email,
  p.phone,
  p.website,
  p.city,
  p.country_code,
  CASE
    WHEN p.country_code = 'CO' THEN CONCAT_WS(', ', NULLIF(p.city, ''), 'Colombia')
    ELSE CONCAT_WS(', ', NULLIF(p.city, ''), NULLIF(p.state_name, ''))
  END AS location,
  ap.logo_letter,
  ap.logo_bg_color,
  ap.image_url,
  ap.summary,
  ap.highlight_review,
  ap.rating,
  ap.reviews_count,
  ap.price_level,
  ap.starting_price,
  ap.map_x,
  ap.map_y,
  ap.distance_km,
  ap.is_verified,
  ap.is_top_rated,
  ap.status,
  ap.slug,
  ap.employees_range,
  ap.experience_years,
  ap.languages,
  ap.work_modes,
  ap.certification_level,
  ap.badge_label,
  ap.recommended,
  ap.commercial_summary,
  ap.case_study,
  ap.response_time_hours,
  ap.qualified_projects,
  ap.trust_score,
  ap.success_rate,
  ap.speciality,
  ap.budget_min,
  ap.budget_max,
  ap.audited,
  ap.profile_completeness
FROM res_partner p
JOIN seo_local_agency_profile ap ON ap.partner_id = p.id
WHERE p.active = TRUE AND ap.status = 'published';
